#define buzzer_pin 8 //定义蜂鸣器引脚

/*------------定义歌曲音调的数组------------*/

int notes[] = {
C5, C5, G5, G5, A5, A5, G5,
F5, F5, E5, E5, D5, D5, C5,
G5, G5, F5, F5, E5, E5, D5,
G5, G5, F5, F5, E5, E5, D5,
C5, C5, G5, G5, A5, A5, G5,
F5, F5, E5, E5, D5, D5, C5,
};

/*------------定义歌曲节奏的数组------------*/

byte beats[] = {
200, 200, 200, 200, 200, 200, 400,
200, 200, 200, 200, 200, 200, 400,
200, 200, 200, 200, 200, 200, 400,
200, 200, 200, 200, 200, 200, 400,
200, 200, 200, 200, 200, 200, 400,
200, 200, 200, 200, 200, 200, 400,
};

void setup() {
pinMode(buzzer_pin, OUTPUT); //定义蜂鸣器引脚为输出状态
}

void loop() {
for ( int i = 0; i < 42; i++) {

/*------------设置i在歌曲长度内递增，这样就能逐一执行数组中的数据------------*/

tone(buzzer_pin, notes[i]); //播放音乐
delay(beats[i]*10); //播放间隔

noTone(buzzer_pin);
delay(10);
}
}
